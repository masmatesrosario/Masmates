    <script>
        const products = [
            { id: 1, name: "Mate Imperial Bordo", price: 32500, emoji: "🧉", cat: "Mates" },
            { id: 2, name: "Termo Acero Gris", price: 45000, emoji: "🍶", cat: "Termos" },
            { id: 3, name: "Set Matero Premium", price: 58000, emoji: "🎁", cat: "Sets" },
            { id: 4, name: "Bombilla Alpaca", price: 12400, emoji: "🪄", cat: "Accesorios" },
            { id: 5, name: "Mate Torpedo Gris", price: 28900, emoji: "🏉", cat: "Mates" },
            { id: 6, name: "Matera de Cuero", price: 22000, emoji: "💼", cat: "Accesorios" },
            { id: 7, name: "Yerba Orgánica 1kg", price: 6500, emoji: "🍃", cat: "Yerba" },
            { id: 8, name: "Mate de Madera", price: 14500, emoji: "🪵", cat: "Mates" }
        ];

        let cart = [];

        function renderProducts() {
            const grid = document.getElementById('product-grid');
            grid.innerHTML = products.map(p => `
                <div class="product-card bg-brand-light-gray/30 rounded-3xl p-6 border border-gray-100 flex flex-col h-full">
                    <div class="w-full aspect-square bg-white rounded-2xl mb-6 flex items-center justify-center text-6xl select-none shadow-sm">
                        ${p.emoji}
                    </div>
                    <span class="text-[10px] font-black text-brand-bordo uppercase tracking-widest mb-1">${p.cat}</span>
                    <h3 class="font-poppins font-bold text-lg text-brand-dark-gray mb-4">${p.name}</h3>
                    <div class="mt-auto flex items-center justify-between">
                        <span class="font-bold text-xl text-brand-bordo">$${p.price.toLocaleString('es-AR')}</span>
                        <button onclick="addToCart(${p.id})" class="w-10 h-10 bg-brand-bordo text-white rounded-full flex items-center justify-center hover:scale-110 transition-transform shadow-md">
                            <svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M12 4v16m8-8H4" />
                            </svg>
                        </button>
                    </div>
                </div>
            `).join('');
        }

        function addToCart(id) {
            const product = products.find(p => p.id === id);
            const inCart = cart.find(item => item.id === id);
            if (inCart) inCart.qty++;
            else cart.push({...product, qty: 1});
            updateCartUI();
            showToast(`Añadido: ${product.name}`);
        }

        function updateCartUI() {
            const cartItems = document.getElementById('cart-items');
            const cartCount = document.getElementById('cart-count');
            const cartTotal = document.getElementById('cart-total');
            const totalQty = cart.reduce((acc, item) => acc + item.qty, 0);
            const totalPrice = cart.reduce((acc, item) => acc + (item.price * item.qty), 0);

            cartCount.textContent = totalQty;
            cartTotal.textContent = `$${totalPrice.toLocaleString('es-AR')}`;

            if (cart.length === 0) {
                cartItems.innerHTML = `<p class="text-center text-gray-400 mt-10 uppercase text-xs font-bold tracking-widest">Carrito Vacío</p>`;
            } else {
                cartItems.innerHTML = cart.map(item => `
                    <div class="flex items-center gap-4 bg-brand-light-gray p-4 rounded-2xl">
                        <div class="text-3xl">${item.emoji}</div>
                        <div class="flex-1">
                            <h4 class="font-bold text-sm text-brand-dark-gray">${item.name}</h4>
                            <div class="text-xs text-brand-bordo font-bold">$${(item.price * item.qty).toLocaleString('es-AR')}</div>
                        </div>
                        <div class="flex items-center gap-2">
                            <button onclick="changeQty(${item.id}, -1)" class="w-6 h-6 bg-white rounded-full border flex items-center justify-center text-xs">-</button>
                            <span class="text-xs font-bold w-4 text-center">${item.qty}</span>
                            <button onclick="changeQty(${item.id}, 1)" class="w-6 h-6 bg-white rounded-full border flex items-center justify-center text-xs">+</button>
                        </div>
                    </div>
                `).join('');
            }
        }

        function changeQty(id, delta) {
            const item = cart.find(i => i.id === id);
            if (!item) return;
            item.qty += delta;
            if (item.qty <= 0) cart = cart.filter(i => i.id !== id);
            updateCartUI();
        }

        function toggleCart() { document.body.classList.toggle('cart-open'); }

        function showToast(msg) {
            const toast = document.getElementById('toast');
            toast.textContent = msg;
            toast.classList.remove('opacity-0');
            setTimeout(() => toast.classList.add('opacity-0'), 2000);
        }

        function checkout() {
            if (cart.length === 0) return showToast("Carrito Vacío");
            showToast("Procesando pedido...");
        }

        document.addEventListener('DOMContentLoaded', renderProducts);
    </script>
